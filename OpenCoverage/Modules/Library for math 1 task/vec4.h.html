#pragma once

{

	vec4() = default;
	vec4(float);
	vec4(float, float, float, float);
	vec4(const vec4&);
	vec4 operator+(const vec4&) const;
	vec4& operator+=(const vec4&);
	vec4 operator-(const vec4&) const;
	vec4& operator-=(const vec4&);

	vec4 operator*(float) const;
	vec4& operator*=(float);
	vec4 operator/(float) const;
	vec4& operator/=(float);

	vec4 Normalize();
